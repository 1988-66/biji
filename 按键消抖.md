int Key_Get(void)

{

  int temp = 1;
  
 if(HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_9) == GPIO_PIN_RESET)
  {
    HAL_Delay(20);
    
  if(HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_9) == GPIO_PIN_RESET)

   {
    while(HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_9) == GPIO_PIN_RESET)

    {

      HAL_Delay(10);

    }

     temp = 0;

   }

  }

  return temp;

}
